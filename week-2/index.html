							const email = "test@profesi.io";
							const subject = "Hello";
							const message = "testing isi email";
							const url = "https://staging-backend.profesi.io/email";
							const body = { email, subject, message };

							//submit request ke Backend untuk diproses
							fetch(url, {
								method: "POST",
								headers: { "Content-Type": "application/json" },
								body: JSON.stringify(body),
							})
							.then((response) => response.json())
							.then((data) => {
								console.log(data)
